CREATE TABLE Cliente (
    ID_Cliente VARCHAR(5),
    RG INT,
    CPF BIGINT,
    Saldo DECIMAL(10, 2),
    Email VARCHAR(255),
    Nome VARCHAR(255),
    AvaliacaoCliente DECIMAL(3, 1),
    DataNasc DATE,
    Idade INT,
    Foto VARCHAR(255)
);

CREATE TABLE AvaliacaoMotorista (
    ID_Motorista VARCHAR(6),
    ID_Cliente VARCHAR(5),
    NotaCliente DECIMAL(3, 1),
    PRIMARY KEY (ID_Motorista, ID_Cliente),
    FOREIGN KEY (ID_Cliente) REFERENCES Cliente(ID_cliente)
);

CREATE TABLE ContatoCliente (
    ID_Cliente VARCHAR(5),
    Telefone VARCHAR(15),
    PRIMARY KEY (ID_cliente)
);

CREATE TABLE Pedido (
    ID_Pedido VARCHAR(5),
    ID_Cliente VARCHAR(5),
    ID_Motorista VARCHAR(6),
    Preco DECIMAL(6, 2),
    TipoDePagamento VARCHAR(20),
    Horario TIME,
    TipoPedido VARCHAR(20),
    PRIMARY KEY (ID_Pedido),
    FOREIGN KEY (ID_cliente) REFERENCES Cliente(ID_cliente),
    FOREIGN KEY (ID_Motorista) REFERENCES AvaliacaoMotorista(ID_Motorista)
);
